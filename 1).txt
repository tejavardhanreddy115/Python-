# Read numbers until -1 is encountered
print("Enter numbers (-1 to exit):")

pos_sum = neg_sum = 0
pos_count = neg_count = 0

while True:
    num = int(input())
    if num == -1:
        break
    if num > 0:
        pos_sum += num
        pos_count += 1
    elif num < 0:
        neg_sum += num
        neg_count += 1

if pos_count > 0:
    avg_pos = pos_sum / pos_count
    print("Average of positive numbers =", int(avg_pos))
else:
    print("No positive numbers entered.")

if neg_count > 0:
    avg_neg = neg_sum / neg_count
    print("Average of negative numbers =", int(avg_neg))
else:
    print("No negative numbers entered.")


output:
Average of positive numbers = 8
Average of negative numbers = -5
